<script>
(function(){
  const loading = document.getElementById('loading');
  const bar = document.getElementById('bar');
  const main = document.getElementById('mainContent');
  const paywall = document.getElementById('paywall');
  const preds = document.getElementById('predictions');
  const refreshBtn = document.getElementById('refreshBtn');
  const payNow = document.getElementById('payNow');
  const emailInput = document.getElementById('emailInput');
  const amountInput = document.getElementById('amountInput');
  const currencySelect = document.getElementById('currencySelect');
  const STORAGE_KEY = 'soc_unlocked_v1';

  function progressSequence(){
    bar.style.width = '12%';
    setTimeout(()=> bar.style.width = '36%', 420);
    setTimeout(()=> bar.style.width = '64%', 1000);
    setTimeout(()=> bar.style.width = '86%', 1600);
    setTimeout(()=> bar.style.width = '100%', 2200);
  }

  async function playIntroSounds(){
    try{
      const ctx = new (window.AudioContext || window.webkitAudioContext)();
      const whistle = ctx.createOscillator();
      const gainW = ctx.createGain();
      whistle.type = 'sine';
      whistle.frequency.setValueAtTime(800, ctx.currentTime);
      whistle.frequency.exponentialRampToValueAtTime(1600, ctx.currentTime + 0.45);
      gainW.gain.setValueAtTime(0, ctx.currentTime);
      gainW.gain.linearRampToValueAtTime(0.12, ctx.currentTime + 0.02);
      gainW.gain.exponentialRampToValueAtTime(0.0001, ctx.currentTime + 0.6);
      whistle.connect(gainW).connect(ctx.destination);
      whistle.start(ctx.currentTime);
      whistle.stop(ctx.currentTime + 0.6);
      await new Promise(r=>setTimeout(r, 700));
      const bufferSize = 2 * ctx.sampleRate;
      const noiseBuffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
      const output = noiseBuffer.getChannelData(0);
      for (let i = 0; i < bufferSize; i++) output[i] = (Math.random()*2-1) * Math.exp(-i/bufferSize*3);
      const noise = ctx.createBufferSource();
      noise.buffer = noiseBuffer;
      const noiseGain = ctx.createGain();
      noiseGain.gain.setValueAtTime(0.0005, ctx.currentTime);
      noiseGain.gain.linearRampToValueAtTime(0.08, ctx.currentTime + 0.08);
      noiseGain.gain.exponentialRampToValueAtTime(0.0001, ctx.currentTime + 2.8);
      const band = ctx.createBiquadFilter();
      band.type = 'bandpass';
      band.frequency.value = 600;
      band.Q.value = 0.8;
      noise.connect(band).connect(noiseGain).connect(ctx.destination);
      noise.start();
      noise.stop(ctx.currentTime + 2.8);
    }catch(e){}
  }

  function revealContent(){
    loading.style.opacity = '0';
    loading.style.pointerEvents = 'none';
    loading.style.transition = 'opacity 420ms ease';
    main.style.opacity = '1';
    main.style.transform = 'translateY(0)';
    main.setAttribute('aria-hidden','false');
    setTimeout(()=> loading.remove(), 600);
  }

  function isoDate(){ const d=new Date(); return d.toISOString().slice(0,10) }

  function isUnlockedToday(){
    try{
      const raw = localStorage.getItem(STORAGE_KEY);
      if(!raw) return false;
      const obj = JSON.parse(raw);
      return obj && obj.date === isoDate();
    }catch(e){ return false }
  }

  function markUnlocked(ref){
    const obj = { date: isoDate(), reference: ref, ts: Date.now() };
    localStorage.setItem(STORAGE_KEY, JSON.stringify(obj));
  }

  function showPredictionsIfUnlocked(){
    if(isUnlockedToday()){
      paywall.style.display = 'none';
      preds.style.display = 'block';
    } else {
      paywall.style.display = 'block';
      preds.style.display = 'none';
    }
  }

  refreshBtn.addEventListener('click', ()=> window.location.reload());

  function toSmallestUnit(amount){
    return Math.round(Number(amount) * 100);
  }

  function pickAmountByCountry(code){
    if(!code) return {currency:'GHS',amount:3};
    const c = String(code).toUpperCase();
    if(c === 'GH' || c === 'GHA') return {currency:'GHS',amount:3};
    if(c === 'NG' || c === 'NGA') return {currency:'NGN',amount:300};
    if(c === 'ZA' || c === 'ZAF') return {currency:'ZAR',amount:8};
    return {currency:'USD',amount:0.25};
  }

  const currencyDefaults = { GHS:3, NGN:300, ZAR:8, USD:0.25 };

  async function detectCountry(){
    try{
      const resp = await fetch('https://ipapi.co/json/');
      if(!resp.ok) return null;
      const j = await resp.json();
      return j.country || j.country_code || j.country_code2 || null;
    }catch(e){ return null }
  }

  async function setupCurrency(){
    currencySelect.innerHTML = '';
    const detected = await detectCountry();
    const picked = pickAmountByCountry(detected);
    const opts = [picked.currency,'GHS','NGN','ZAR','USD'];
    const unique = Array.from(new Set(opts));
    unique.forEach(c=>{
      const o = document.createElement('option');
      o.value = c;
      o.textContent = c;
      currencySelect.appendChild(o);
    });
    currencySelect.value = picked.currency;
    amountInput.value = currencyDefaults[picked.currency] !== undefined ? currencyDefaults[picked.currency] : picked.amount;
  }

  currencySelect.addEventListener('change', ()=>{
    const val = currencySelect.value || 'GHS';
    amountInput.value = currencyDefaults[val] !== undefined ? currencyDefaults[val] : amountInput.value;
  });

  function openPaystackInline(email, amount, currency){
    const handler = PaystackPop.setup({
      key: 'pk_live_yourpublickeyhere',
      email: email,
      amount: toSmallestUnit(amount),
      currency: currency,
      onClose: function(){ alert('Payment window closed.'); },
      callback: function(response){
        markUnlocked(response.reference || response.trxref || response.transaction);
        showPredictionsIfUnlocked();
        alert('Payment successful. Reference: ' + (response.reference||response.trxref||response.transaction));
      }
    });
    handler.openIframe();
  }

  payNow.addEventListener('click', ()=>{
    const email = (emailInput.value||'').trim();
    const amount = (amountInput.value||'').trim();
    const currency = currencySelect.value || 'GHS';
    if(!email || !/\S+@\S+\.\S+/.test(email)){
      alert('Enter a valid email before paying.');
      return;
    }
    if(!amount || Number(amount) <= 0){
      alert('Enter a valid amount.');
      return;
    }
    openPaystackInline(email, amount, currency);
  });

  document.addEventListener('DOMContentLoaded', ()=>{
    main.style.transition = 'opacity 420ms ease, transform 420ms ease';
    main.style.opacity = '0'; main.style.transform='translateY(12px)';
    progressSequence();
    playIntroSounds();
    setupCurrency().then(()=>{ setTimeout(()=>{ revealContent(); showPredictionsIfUnlocked(); }, 3000); });
  });

  document.addEventListener('keydown', (e)=> { if(e.key==='r' || e.key==='R'){ window.location.reload() } });

})();
</script>
